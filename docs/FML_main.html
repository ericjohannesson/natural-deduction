val fml_of_string : ?print_tokens:bool -> string -> FML_types.t_fml

fml_of_string fml evaluates to an object of type FML_types.t_fml if fml conforms to the grammar specified below, and raises exception Parse_error fml otherwise.

fml ::=
        | UNOP fml
        | QUANT VAR fml
        | fml BINOP fml
        | atomic_fml
        | LPAR fml RPAR
        | LBR fml RBR

atomic_fml ::=
        | PREFIX_PRED
        | PREFIX_PRED par_terms
        | term INFIX_PRED term
        | term NEG_INFIX_PRED term

par_terms ::=
        | LPAR terms RPAR
        | LBR terms RBR

terms ::=
        | term
        | term COMMA terms

term ::=
        | VAR
        | PREFIX_FUNC
        | PREFIX_FUNC par_terms
        | term INFIX_FUNC term
        | term POSTFIX_FUNC
        | LPAR term RPAR
        | LBR term RBR


UNOP ::=
        | "¬" | "\neg" | "~"

QUANT ::=
        | "∀" | "\forall" | "\A"
        | "∃" | "\exists" | "\E"

BINOP ::=
        | "→" | "\to"   | "\rightarrow" | "->"
        | "∧" | "\land" | "\wedge"      | "&"
        | "∨" | "\lor"  | "\vee"
        | "↔" | "\leftrightarrow"       | "<->"

LPAR ::= "("

RPAR ::= ")"

LBR ::= "["

RBR ::= "]"

COMMA ::= ","

VAR ::= ['u' - 'z'] (['_' '.'] | ['a' - 'z'] | ['0' - '9'])*

PREFIX_PRED ::= ['A' - 'Z'] (['_' '.'] | ['A' - 'Z'] | ['a' - 'z'] | ['0' - '9'])*

INFIX_PRED ::=
        | "="
        | "<"
        | ">"
        | "≤" | "\leq" 
        | "≥" | "\geq"
        | "∈" | "\in"
        | "⊂" | "\subset"
        | "⊆" | "\subseteq"

NEG_INFIX_PRED ::=
        | "≠" | "\neq" 
        | "∉" | "\not\in" | "\not \in" | "\notin"

PREFIX_FUNC ::=
        | ['a' - 't'] (['_' '.'] | ['a' - 'z'] | ['0' - '9'])*
        | ['0' - '9']+

INFIX_FUNC ::=
        | "+"
        | "×" | "\times" 
        | "*"
        | "·" | "\cdot" 
        | "-"
        | "∩" | "\cap" 
        | "∪" | "\cup"
        | "^"

POSTFIX_FUNC ::=
        | "'"
        | "²"
        | "³"

subst_in_fml var term fml evaluates to the result of replacing every free occurrence of var in fml with term, regardless of whether term contains variables that become bound.
