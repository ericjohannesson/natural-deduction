let fml_list_of_file (path : string) : t_fml list = let ic = open_in path in let rec aux (acc : FML_types.t_fml list) : t_fml list = try let s : string = input_line ic in let fml : t_fml = fml_of_string s in aux (fml::acc) with |End_of_file -> acc |Parse_error e -> let _ : unit = IO.print_to_stderr e in acc in List.rev (aux [])
